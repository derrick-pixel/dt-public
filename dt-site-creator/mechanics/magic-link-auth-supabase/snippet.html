-- Sanitized: real domain values replaced with placeholders.

-- CRITICAL: This hook is mandatory — without it, signInWithOtp() creates the
-- user row at magic-link click time regardless of domain. App-layer checks are
-- guidance, not enforcement. This Postgres hook is the actual gate.
--
-- Wire-up: Supabase Dashboard → Auth → Hooks → "Before User Created"
--   Function: public.hook_restrict_signup_by_email_domain

-- 1. Domain allowlist table
create table if not exists public.signup_email_domains (
  domain text primary key
);

-- 2. Add your authorized domains (replace placeholders)
insert into public.signup_email_domains (domain) values
  (''),   -- e.g. 'yourcompany.com'
  ('')    -- e.g. 'partner.com' — remove line if only one domain
on conflict (domain) do nothing;

-- 3. Hook function — rejects signups whose email domain is not in the table
create or replace function public.hook_restrict_signup_by_email_domain(event jsonb)
returns jsonb
language plpgsql
security definer
set search_path = ''
as $$
declare
  email_domain text;
begin
  -- Extract domain from email, handling all event shapes Supabase may send
  email_domain := lower(split_part(coalesce(
    event->'user'->>'email',
    event->'user_metadata'->>'email',
    event->>'email'
  ), '@', 2));

  -- Reject if no email was present
  if email_domain is null or email_domain = '' then
    return jsonb_build_object(
      'error', jsonb_build_object(
        'http_code', 403,
        'message', 'No email present on signup request.'
      )
    );
  end if;

  -- Allow if domain is in the allowlist
  if exists (select 1 from public.signup_email_domains where domain = email_domain) then
    return jsonb_build_object('decision', 'continue');
  end if;

  -- Reject all other domains
  return jsonb_build_object(
    'error', jsonb_build_object(
      'http_code', 403,
      'message', 'This site is restricted to authorized email domains.'
    )
  );
end;
$$;

-- 4. Permissions — supabase_auth_admin must execute; everyone else blocked
grant execute on function public.hook_restrict_signup_by_email_domain to supabase_auth_admin;
revoke execute on function public.hook_restrict_signup_by_email_domain from authenticated, anon, public;

grant select on table public.signup_email_domains to supabase_auth_admin;
revoke all on table public.signup_email_domains from authenticated, anon, public;
