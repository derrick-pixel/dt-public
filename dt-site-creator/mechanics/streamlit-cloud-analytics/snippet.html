Guest rate-limiter pattern (st.session_state + SQLite) ```python # guest_limit.py — IP-based daily time limiter (extracted from market-tracker, sanitized). # # Tracks wall-clock active time per IP per calendar day using SQLite. # "Active" = a rerun fired within IDLE_THRESHOLD seconds. # Long idle gaps are not counted, so the budget measures real usage. # # ⚠ SQLite works on Streamlit Cloud because the instance has ephemeral local # disk. The counter resets if the app spins down and is restarted — good # enough for casual abuse prevention, not for hard enforcement. # # Usage in app.py: # import guest_limit # st.set_page_config(...) # must be first # guest_limit.enforce() # second line — blocks blocked users immediately import sqlite3, time from datetime import date, datetime, timezone from pathlib import Path import streamlit as st LIMIT_SECONDS = 600 # 10 minutes per IP per day IDLE_THRESHOLD = 120 # gaps > 2 min not counted as active time DB_PATH = Path(__file__).parent / ".guest_usage.db" EXEMPT_IPS = {"127.0.0.1", "::1", "localhost"} def _conn() -> sqlite3.Connection: c = sqlite3.connect(DB_PATH, timeout=5) c.execute("PRAGMA journal_mode=WAL") return c def _get_ip() -> str: try: h = st.context.headers fwd = h.get("X-Forwarded-For", "") if fwd: return fwd.split(",")[0].strip() except Exception: pass return "127.0.0.1" # fallback → always exempt def _tick(ip: str, today: str) -> float: """Record one active tick; return cumulative seconds used today.""" with _conn() as c: c.execute(""" CREATE TABLE IF NOT EXISTS ip_usage ( ip TEXT NOT NULL, date TEXT NOT NULL, seconds_used REAL NOT NULL DEFAULT 0, PRIMARY KEY (ip, date) )""") now = time.monotonic() if "gl_last_tick" in st.session_state: delta = now - st.session_state.gl_last_tick if 0 < delta < IDLE_THRESHOLD: c.execute(""" INSERT INTO ip_usage VALUES (?,?,?) ON CONFLICT(ip,date) DO UPDATE SET seconds_used = seconds_used + excluded.seconds_used """, (ip, today, delta)) st.session_state.gl_last_tick = now row = c.execute( "SELECT seconds_used FROM ip_usage WHERE ip=? AND date=?", (ip, today) ).fetchone() return float(row[0]) if row else 0.0 def enforce() -> None: """Call once in app.py right after st.set_page_config().""" ip, today = _get_ip(), str(date.today()) if ip in EXEMPT_IPS: return used = _tick(ip, today) if used >= LIMIT_SECONDS: now_utc = datetime.now(timezone.utc) midnight = now_utc.replace(hour=23, minute=59, second=59) hh, mm = divmod(int((midnight - now_utc).total_seconds()) // 60, 60) st.error(f"Daily guest limit reached. Resets in {hh}h {mm}m (midnight UTC).") st.stop() ```
